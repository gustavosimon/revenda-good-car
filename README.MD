`Comandos SQL para criaçã do banco`

```bash

# Create table MARCA
CREATE TABLE MARCA(ID INT not null primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), NOME_MARCA VARCHAR(30))

# Create table CLIENTE
CREATE TABLE CLIENTE(ID INT not null primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), NOME VARCHAR(40), ENDERECO VARCHAR(60), TELEFONE BIGINT, EMAIL VARCHAR(40))

# Create table VEICULO
CREATE TABLE VEICULO(ID INT not null primary key GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), MODELO VARCHAR(40), ANO INT, QUILOMETRAGEM BIGINT, POTENCIA BIGINT, PRECO BIGINT)

# Add the foreign key
ALTER TABLE SIMON.VEICULO ADD FOREIGN KEY (ID_MARCA) REFERENCES SIMON.MARCA(ID)

```
